    <script>

        const ADMIN_PASSWORD = "Kheda@2026";   // ← Change this if you want

        let submissions = JSON.parse(localStorage.getItem('itiSubmissions')) || [];

        document.getElementById('registrationForm').addEventListener('submit', function(e) {

            e.preventDefault();

            const pass = document.getElementById('traineePassword').value;

            if (pass.length < 6 || !/[0-9]/.test(pass) || !/[!@#$%^&*]/.test(pass)) {

                alert("Password must contain letters, numbers and symbols (min 6 characters)");

                return;

            }

            const formData = {

                Date: new Date().toLocaleDateString('en-IN'),

                FullName: document.getElementById('fullName').value,

                FatherName: document.getElementById('fatherName').value,

                DOB: document.getElementById('dob').value,

                Mobile: document.getElementById('mobile').value,

                Trade: document.getElementById('trade').value,

                TraineePassword: document.getElementById('traineePassword').value,

                Status: "Pending"

            };

            submissions.push(formData);

            localStorage.setItem('itiSubmissions', JSON.stringify(submissions));

            document.getElementById('page1').classList.remove('active');

            document.getElementById('page2').classList.add('active');

        });

        function showAdminLogin() {

            document.getElementById('page2').classList.remove('active');

            document.getElementById('adminPage').classList.add('active');

        }

        function verifyAdmin() {

            if (document.getElementById('adminPass').value === ADMIN_PASSWORD) {

                alert("✅ Welcome Admin! You can now download the Excel sheet.");

            } else {

                alert("❌ Wrong Password");

            }

        }

        function downloadExcel() {

            if (document.getElementById('adminPass').value !== ADMIN_PASSWORD) {

                alert("Please enter correct admin password first");

                return;

            }

            if (submissions.length === 0) {

                alert("No registrations yet");

                return;

            }

            const ws = XLSX.utils.json_to_sheet(submissions);

            const wb = XLSX.utils.book_new();

            XLSX.utils.book_append_sheet(wb, ws, "All_Registrations");

            XLSX.writeFile(wb, "SRIJAN_ITI_All_Registrations.xlsx");

        }
